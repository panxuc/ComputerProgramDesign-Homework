```c
#include <stdio.h>

void input(int n, char **p)
{

    {
        scanf("%s", str);
        p[i] = (char*)malloc(sizeof(strlen(str)));
        strcpy(p[i], str);
    }
}
void sort(int n, char **p)
{
    char *t;
    int i = 0, j = 0, k = 0;
    for(k = 79; k >= 0; k--)
    {
        for(i = 0; i < n - 1; i++)
        {
            for(j = 0; j < n - 1 - i; j++)
            {
                if(p[j][k] > p[j + 1][k])
                {
                    t = p[j];
                    p[j] = p[j + 1];
                    p[j + 1] = t;
                }
            }
        }
    }
}
int main()
{
    int n, i;
    scanf("%d", &n);
    char **p = (char**)malloc(sizeof(char*) * n);
    input(n, p);
    sort(n, p);
    printf("\n\n");
    for(i = 0; i < n; i++)
    {
        printf("%s\n", p[i]);
    }

}
```

### 必做第2题

程序代码
```c
#include <stdio.h>
#define N 10
int maruplus(int i, int n)
{
    int ret;
    if(i + n <= N) ret = i + n;
    else ret = (i + n) % N;
    return ret;
}
int main()
{
    int a[10][10];
    int *p;
    int sum = 0, i = 1, n;

    for(p = (int*)a; p < (int*)a + 100; p++)
    {
        *p = p - (int*)a;
    }
    scanf("%d", &n);
    for(i = 1; i <= N; i++)
    {
        sum += *((int*)a + (i - 1) * N + maruplus((i - 1), n));
    }
    printf("%d", sum);
    return 0;
}
```

### 必做第3题

```c
#include <stdio.h>
#include <string.h>
int main()
{
    char tweet[140] = {0};
    char charBegin, charEnd;
    char *emojiBegin, *emojiEnd, *p;
    int i = 0, hasBegin = 0, hasEnd = 0;
    printf("转义符：");
    scanf("%c", &charBegin);
    getchar();
    printf("终止符：");
    scanf("%c", &charEnd);
    getchar();
    printf("输入文字：");
    gets(tweet);
    for(i = 0; i < strlen(tweet); i++)
    {
        if(*(tweet + i) == charBegin && !hasEnd)
        {
            emojiBegin = tweet + i;
            hasBegin = 1;
        }
        if(*(tweet + i) == charEnd && hasBegin)
        {
            emojiEnd = tweet + i;
            hasEnd = 1;
            break;
        }
    }
    printf("输出：");
    for(p = emojiBegin + 1; p < emojiEnd; p++)
        printf("%c", *p);
    return 0;
}
```
